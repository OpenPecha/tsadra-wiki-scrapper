༡༧༡ ༈ །།སེམས་སྡེ་བཅོ་བརྒྱད་ཀྱི་དགོངས་པ་རིག་འཛིན་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུར་བཞེངས་པ་བཞུགས་སོ། །ན་མོ་གུ་རུ་བིདྱཱ་དྷཱ་རེ་བྷྱཿ དེ་ལ་ཁྱུང་ཆེན་གྱི་དགོངས་པ། ཆོས་ཉིད་ཀློང་རྡོལ་གླུ་ཡི་བསམ་གཏན། ཙནྡན་གྱི་ནགས་ཚལ་དུ་ནམ་གྱི་གུང་ལ། རྒྱལ་པོ་དྷ་ཧེ་ན་ཏེ་ལོ་འདི་སྐད་ལོ། །ཧཱུྃ། འགྱུར་བ་མེད་པའི་ཆོས་ཉིད་ནི། །ཀུན་གཞི་མ་བཅོས་བྱང་ཆུབ་སེམས། །གཟུང་འཛིན་རྣམ་རྟོག་ཡུལ་ལས་འདས། །སྣང་བའི་གཟུགས་མེད་བྱང་ཆུབ་སེམས། །ཅི་ལའང་མི་སྤྱད་རྟོག་པས་མི་བཟུང་ཞིང་། །སེམས་ཉིད་རང་བཞིན་བསམ་མེད་མཁའ་འདྲ་ལ། །མ་ཡེངས་མ་བསླད་བཅོས་པ་མེད་པར་བཞག །སྒོམ་དུ་མེད་པའི་ཆོས་དབྱིངས་ཆེན་པོ་ལ། །བསམ་དུ་མེད་པའི་རིག་པ་མ་ཡེངས་ཞོག །དེ་ལ་སྒོམ་རྒྱུ་རྡུལ་ཙམ་མེད་པར་འཁུམས། །སྐྱེ་བ་བརྒྱུད་ནས་རྒྱལ་པོར་བདག་སྐྱེས་ཏེ། །དོ་ཧ་དེ་བླངས་ང་རྒྱལ་བཅག་པ་ལས། །སྔོན་ཆད་བྱང་ཆུབ་མཆོག་ཏུ་སེམས་བསྐྱེད་ཅིང་། །སེམས་དང་ཆོས་སུ་འདྲེས་པའི་འཕྲོ་བླངས་ཞིང་། །སྒྱུ་མའི་ཕུང་པོ་ད་ལྟ་བསྙིལ་བའི་ཕྱིར། །ཀུན་བཟང་ཐུགས་ཀྱི་འོད་ཟེར་བདག་ལ་ཤར། །གྷི་ཏ་མ་ཀོ་ས་ལ་ཨ་ཧོཿ ཞེས་སོ། །དེ་སྐད་དུ་ཡང་དྲང་སྲོང་བྷ་ཤིས། །སེམས་ནི་ཅི་ལ་ཡང་མི་རྟོག་པ། །མ་བཅོས་པ་དེ་རྣལ་མར་ཞོག །ཅེས་སོ། །ཁྱུང་ཆེན་གྱི་དགོངས་པ་དེ་དང་པོའོ། །། ༈ །རིག་པའི་ཁུ་བྱུག་གི་དགོངས་པ། རིག་པ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། ཙནྡན་གོི་ཤའི་ཚལ་དུ་ནམ་གྱི་ཆ་སྟོད་ལ། སྲས་ཐུ་པོ་རཱ་ཛ་ཧ་སྟི་འདི་སྐད་ལོ། །ཧཱུྃ། ཀུན་གཞི་ཡེ་ནས་མ་བཅོས་དེ་བཞིན་ཆོས་ཀྱི་དབྱིངས། །རང་བྱུང་ཡེ་ཤེས་མཁའ་འདྲ་མཉམ་པ་ཆེན་པོའི་
༡༧༢དང་། །ཚིག་དང་ཡི་གེའི་སྒྲ་ལས་འདས་པ་ཀུན་ཏུ་བཟང་། །སེམས་དང་ཆོས་ཉིད་མཉམ་ཞིང་དབྱེར་མེད་ན། །ལུས་ངག་ཡིད་གསུམ་མཚན་མ་ཅིས་སླད་ཀྱང་། །ཆོས་ཉིད་བྱང་ཆུབ་སེམས་ལ་འགྱུར་མེད་དེ། །རྣམ་རྟོག་སྐྱོན་སྤངས་ཀུན་གཞི་བྱང་ཆུབ་སེམས། །མ་བཅོས་ངང་དུ་བཞག་པ་ལས། །སྒོམ་ཞེས་གུད་ན་མེད་པར་འཁུམས། །སྒྱུ་མའི་ཕུང་པོ་ད་ལྟ་བསྙིལ་བའི་ཕྱིར། །ཀུན་བཟང་ཐུགས་ཀྱི་འོད་ཟེར་ང་ལ་ཤར། །གྷི་ཏ་མ་ཀོ་ལ་ས་མ་ཧོཿ ཞེས་གསུང་། གཞན་ཡང་ཤྲཱི་སེང་ཧས། ཡེ་ཤེས་སྣ་ཚོགས་རང་གསལ་ལ། །མི་གཡོ་དེ་བཞིན་ཉིད་དུ་ཞོག །ཅེས་སོ། །རིག་པའི་ཁུ་བྱུག་གི་དགོངས་པ་སྟེ་གཉིས་པའོ། །།། ༈ །ནམ་མཁའ་ཆེའི་དགོངས་པ། ཡེ་ཤེས་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། མེ་ཏོག་ཨུ་དུམྦཱ་རའི་ཚལ་དུ་ནམ་སྨད་ཀྱི་ཆ་ལ། སྲས་མོ་སཱ་ར་ནིས་འདི་སྐད་ལོ། །ཀུན་གཞི་མ་བཅོས་ཡེ་ནས་ཆོས་ཀྱི་དབྱིངས། །རྒྱུ་དང་རྐྱེན་གྱིས་མ་བསྐྱེད་འཇིག་བྲལ་མེད། །རང་བྱུང་ལྷུན་གྱིས་གྲུབ་པ་བཅོམ་ལྡན་འདས། །རང་བྱུང་ཡུལ་ལས་འདས་པ་རྡོ་རྗེ་སེམས་དཔའི་ཐུགས། །བདག་གི་དགོངས་པ་འདི་ལས་མེད་པར་འཁུམས། །སྒྱུ་མའི་ཕུང་པོ་ད་ལྟ་བསྙིལ་བའི་ཕྱིར། །ཀུན་བཟང་ཐུགས་ཀྱི་གཏེར་ཆེན་བདག་ལ་བབས། །གྷི་ཏ་མ་ཀོ་ལ་ཧོཿ ཞེས་གསུངས། གཞན་ཡང་། འཕགས་པ་ལྷས། རིག་པ་དེ་ཉིད་དངོས་གཞན་མེད། །ངེས་ན་རྟོག་མེད་གསལ་བར་བཞག །ཅེས་སོ། །ནམ་མཁའ་ཆེའི་དགོངས་བཅུད་དེ་གསུམ་པའོ། །། ༈ ཐིག་ལེ་དྲུག་པའི་དགོངས་པ། ནམ་མཁའ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། སྐྱ་རེངས་དང་པོ་ཤར་ཙམ་ན་རྒྱ་མཚོའི་གཏིང་རུམ་དུ་ཀླུའི་རྒྱལ་པོས་འདི་སྐད་དོ། །མཐའ་དང་དབུས་མེད་ནམ་མཁའ་ལྟ་བུ་ལ། །སྨ་བསམ་ཡུལ་ལས་འདས་པ་
༡༧༣ཆོས་ཀྱི་སྐུ། །སྲིད་འབྱུང་སྐྱེ་འགྲོ་རྣམ་པར་དག་བྱས་ནས། །མ་བཅོས་རྣམ་པར་དག་པས་ཀུན་གཞི་བྱང་ཆུབ་སེམས། །མ་བཙལ་མཉམ་པར་གཞག་ན་བསྒོམ་པའི་མཆོག །རིག་པ་ཆོས་ཀྱི་དབྱིངས་སུ་མ་བཅོས་ལ། །རྣམ་པར་གྲོལ་བའི་ལམ་མཆོག་གཞན་མེད་དེ། །དུས་གསུམ་བདེ་བར་གཤེགས་པའི་དགོངས་པ་ཉིད། །དཔག་ཅིང་གཞལ་ཡང་འདི་ལས་མེད་པར་མཇལ། །བདག་གི་ཕུང་པོ་གདོས་བཅས་འདི་བསྙིལ་ནས། །ཀུན་བཟང་དབྱེ་བ་མེད་པའི་ས་ལ་མཆི། །གྷི་ཏ་མ་ཀོ་ལ་ས་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་མ་ཁོལ་ལུང་བསྟན་ལས། །ཆོས་རྣམས་ཡེ་ནས་སངས་རྒྱས་པས། །གཞན་ལ་རེ་བཙལ་གཉིས་དང་འབྲལ། །ཞེས་སོ། །ཐིག་ལེ་དྲུག་པའི་དགོངས་བཅུད་དེ་བཞི་པའོ། །། ༈ །རྩལ་ཆེན་གྱི་དགོངས་པ། བྱང་སེམས་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། སྐྱ་རེངས་གཉིས་པ་ཤར་ཙམ་ན། རྒྱ་མཚོའི་གླིང་དུ་ཁྱིའི་རྒྱལ་པོ་ཀུ་ཀུ་རཱ་ཛས་འདི་སྐད་ལོ། །ཡེ་ནས་ལྷུན་གྲུབ་ཀུན་ཏུ་བཟང་པོ་དང་། །གསལ་ལ་མི་རྟོག་སྤྲོས་བྲལ་བྱང་ཆུབ་སེམས། །མི་འགྱུར་མི་དམིགས་ཆོས་ཀྱི་དབྱིངས་ཉིད་ལས། །རིག་པ་མི་འགྱུར་མ་བསླད་མཉམ་བཞག་ན། །རྩི་ལའང་མི་རྟོག་གང་ལའང་མི་བསམས་ཏེ། །མཆོག་ཏུ་གསང་བའི་རྨད་བྱུང་སེམས་ཉིད་འདི། །རྣ་བས་ཉན་པ་ལ་སོགས་སྤྱོད་ཡུལ་ཡོད་མ་ཡིན། །ཀུན་གཞི་མི་འགྱུར་དབྱིངས་ཆེན་ལ། །རིག་པ་མི་འགྱུར་མ་ཡེངས་བཞག །སྒོམ་པ་འདི་ལས་མེད་པར་འཁུམས། །ཁམས་གསུམ་རང་སར་གྲོལ་བས་བཙལ་ཞིང་སྒྲུབ་མི་དགོས། །རེ་ཞིང་དོགས་པ་མེད་པས་བདག་ཉིད་སངས་རྒྱས་ཡིན། །གྷི་ཏི་མ་ཀོ་ལ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། ཨཱ་དེ་བས། བཟང་ངན་བླང་དོར་མེད་པས་ཡེ་ཤེས་ཟང་ཐལ་ཏེ། །ཡེ་ཤེས་ཉིད་ལ་ཉིད་སྤྱོད་གསེར་གླིང་འདྲ།
༡༧༤ཞེས་སོ། །རྩལ་ཆེན་གྱི་དགོངས་བཅུད་དེ་ལྔ་པའོ། །། ༈ །སེམས་སྒོམ་གྱི་དགོངས་པ། དབྱིངས་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། རྒྱན་ཆགས་དཔལ་གྱི་རིའི་ནགས་སུ་སྐྱ་རེངས་གསུམ་པ་ཤར་ཙམ་ན། ནཱ་གརྫུ་ནས་འདི་སྐད་དོ། །སྣང་བ་ཐམས་ཅད་ཡེ་ནས་དབྱིངས་ཆེན་ལ། །རིག་པ་འཕོ་འགྱུར་མེད་པའི་དུར་བཏབ་ནས། །དབྱིངས་ཉིད་མི་འགྱུར་རིག་པ་མི་བསླད་དེ། །མ་ཡེངས་མི་འགྱུར་མཁའ་ལྟར་དག་པ་ལ། །མ་ཡེངས་མ་བཅོས་ཡོངས་ཀྱི་འོད་གསལ་ལས། །སྒོམ་པ་མིང་དུ་བཏགས་པ་གུད་ན་མེད་པར་འཁུམས། །བདག་གི་ཕུང་པོ་གདོས་བཅས་འདིས། །དབྱེ་བ་མེད་པའི་ས་ལ་མཆི། །གྷི་ཏ་མ་ཀོ་ལ་ས་ཧོཿ ཞེས་གསུངས། གཞན་ཡང་ཤེལ་ཕྲེང་དཀར་པོ་ཞེས་བྱ་བའི་མན་ངག་ལས། སྣང་སྲིད་ཐམས་ཅད་ཆོས་ཉིད་ཡེ་ནས་ཡིན། །དེས་ན་ཆོས་ཉིད་དེ་ཉིད་དེ་ལ་སྤྱོད། །ཅེས་སོ། །སེམས་སྒོམ་གྱི་དགོངས་བཅུད་དེ་དྲུག་པའོ། །། ༈ །རྨད་བྱུང་གི་དགོངས་པ། འབྱུང་ལྔ་ཡེ་ཤེས་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། དམར་རེངས་དང་པོ་ཤར་ཙམ་ན་རྒྱ་མཚོའི་འགྲམ་དུ་ཁྱིའི་རྒྱལ་པོ་ཕྱི་མས་འདི་སྐད་ལོ། །ཆོས་ཉིད་བྱང་ཆུབ་སེམས་ནི་འབྱུང་བ་ལྔ་ཡིན་ཏེ། །མ་བཅོས་རང་བཞིན་གསལ་བས་མཁའ་ལྟར་གནས། འགྲོ་བའི་གཟུགས་འཛིན་སྲེགས་པས་མེ་ལྟར་གནས། །དམིགས་མེད་གདོད་ནས་དག་པས་རླུང་ལྟར་གནས། །གཟུང་འཛིན་སྐྱོན་དང་བྲལ་བས་ཆུ་ལྟར་གནས། །ཅིས་ཀྱང་སྐྱོན་མེད་སྲ་བས་ས་ལྟར་གནས། །འབྱུང་བ་ལྔ་ཉིད་ཡེ་ནས་ཆོས་ཉིད་བྱང་ཆུབ་སེམས། །གདོད་ནས་དག་པས་ནམ་མཁའ་འདྲ་བ་ལ། །དམིགས་པ་མེད་པར་མ་ཡེངས་གནས་པ་ལས། །སྒོམ་པ་འདི་ལྟར་མེད་པར་རྟོགས། །བདག་གི་ཕུང་པོ་གདོས་བཅས་འདི། །དབྱེ་བ་མེད་པའི་ས་ལ་མཆི། །གྷི་ཏ་མ་ཀོ་ལ་ས་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །
༡༧༥གཞན་ཡང་རྡོ་རྗེ་བརྩེགས་པ་ལས། །རྣམ་རྟོག་ཡེ་ཤེས་ངང་དུ་གསལ། །དེས་ན་མ་སྤངས་རང་ཞིར་ཞོག །ཅེས་སོ། །རྨད་བྱུང་རྒྱལ་པོའི་དགོངས་བཅུད་དེ་བདུན་པའོ། །། ༈ །ཡིད་བཞིན་ནོར་བུའི་དགོངས་པ། དངོས་གྲུབ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། །དམར་རེངས་གཉིས་པ་ཤར་ཙམ་ན་གནོད་སྦྱིན་མོ་བྱང་ཆུབ་འདི་སྐད་ལོ། །ཆོས་ཉིད་བྱང་ཆུབ་སེམས་ལ་བླང་ཞིང་བལྟ་མེད་དོ། །ལྟ་བའི་གཟུགས་སུ་ཡོངས་ལ་སྣང་། །ཅིས་ཀྱང་བཅོས་སུ་མེད་ལ་བཅད་དུ་མེད། །སེམས་ཀྱི་རང་བཞིན་ཅི་ཡང་མི་གནས་ཏེ། །རིག་པ་བྱང་ཆུབ་སེམས་ཀྱི་རང་བཞིན་ཡིན། །ཇི་ལྟར་ལྷད་ཀྱང་དབྱིངས་ལ་འགྱུར་བ་མེད། །དབྱིངས་ཉིད་རིག་པས་མི་བཙལ་ཏེ། །སེམས་ཉིད་མཁའ་ལྟར་དག་པའི་དངོས། །མ་བཅོས་རང་བཞིན་མེད་པ་ལས། །སྒོམ་པ་འདི་ལས་མེད་པར་བདག་གིས་རྟོགས། །བདག་གི་ཕུང་པོ་གདོས་བཅས་འདི་བསྙིལ་ནས། །ཀུན་བཟང་ཐུགས་ཀྱི་འོད་ཟེར་བདག་ལ་ཤར། །གྷི་ཏ་མ་ཀོ་ལ་ས་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་རཱ་ཧུ་ལ་ཆེན་པོས། འཁོར་བ་མ་སྤངས་ངང་གིས་གསལ་བ་ལ། །ཡེ་ཤེས་ཉི་མ་དེ་ཡི་ཚུལ་དུ་འཆར། །ཞེས་སོ། །ཡིད་བཞིན་ནོར་བུའི་དགོངས་བཅུད་དེ་བརྒྱད་པའོ། །། ༈ །རྩེ་མོ་བྱུང་རྒྱལ་གྱི་དགོངས་པ། རྩོལ་མེད་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། ཤིང་ནལ་བྱིས་ལའི་རྩར། དམར་རེངས་གསུམ་པ་ཤར་ཙམ་ན་སྨད་འཚོང་མ་བུདྡྷ་མ་ཏི་འདི་སྐད་ལོ། །ཨེ་མ་ཧོ་ཆོས་ཉིད་བྱང་ཆུབ་སེམས། །བརྟག་ཏུ་མེད་དེ་དཔྱད་དུ་མེད། །བཙལ་ཞིང་གཉེན་པོས་བཅོས་སུ་མེད། །སེམས་ལས་སངས་རྒྱས་ལོགས་ན་མེད། །བཙལ་བས་མི་རྙེད་སེམས་ལས་སངས་རྒྱས་མེད། །སེམས་ཀྱི་རང་བཞིན་མ་བཅོས་སངས་རྒྱས་ཡིན། །གདོད་ནས་མ་བཙལ་བྱང་ཆུབ་སྙིང་པོ་ལ། །ཏིང་འཛིན་
༡༧༦མ་ལྷད་ངང་གིས་གསལ་བ་ལས། །སངས་རྒྱས་དགོངས་པ་འདི་ལས་མེད་པར་འཁུམས། །བདག་གི་ཕུང་པོ་གདོས་བཅས་འདི་བསྙིལ་ནས། །ཀུན་བཟང་ཐུགས་ཀྱི་འོད་ཟེར་བདག་ལ་ཤར། །གྷི་ཏ་ཀོ་ལས་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་གུ་རུ་ཆེན་པོས། ཐམས་ཅད་ཡེ་ཤེས་གསལ་བའི་རྐྱན། །དེ་ལྟར་རྟོགས་པས་དག་པ་ཡིན། །ཅེས་སོ། །རྩེ་མོ་བྱུང་རྒྱལ་གྱི་དགོངས་བཅུད་དེ་དགུ་པའོ། །། ༈ སྤྱི་གཅོད་ཀྱི་དགོངས་པ། གསང་བ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། ཤིང་ནྱ་གྲོ་དྷཱའི་རྩ་བར་སྲོད་ཐུན་ལ་འཇམ་དཔལ་བཤེས་གཉེན་ཕྱི་མ་འདི་སྐད་ལོ། །ཀུན་གཞི་ཡངས་པའི་དབྱེ་ཐང་ལ། །དག་པའི་ཀློང་ནས་ཉི་མ་ཤར། །ཤར་བའི་དུས་ན་ནུབ་པ་མེད། །ཤར་ནས་མ་བཅོས་གསལ་པ་ལས། །སངས་རྒྱས་ཆོས་ཉིད་སྔ་ཕྱི་གསུམ། །གང་དུ་བཙལ་ཡང་རྙེད་ཐབས་མེད། །ཀུན་གཞི་ཡངས་བའི་ཆོས་ཉིད་ལ། །རིག་པ་ཡེ་ཤེས་ཅི་ཤར་བ། །མ་བཅོས་ནུབ་པ་མེད་པ་ལས། །སྒོམ་དུ་འདི་ལས་མེད་པར་འཁུམས། །མྱོང་ན་སེམས་ཀྱི་བདུད་རྩི་མྱོང་། །བལྟ་ན་རིག་པའི་ཡེ་ཤེས་ལྟོས། །པདྨ་ཀོ་ལ་ཤ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། ལུང་ཕྱི་མ་ལས། ཉམས་སུ་མྱོང་བའི་བདེ་ཆེན་ནི། །ཐམས་ཅད་གསལ་བ་ཞེ་ཕུག་ཡིན། །ཅེས་སོ། །རྫོགས་པ་སྤྱི་གཅོད་ཀྱི་དགོངས་བཅུད་དེ་བཅུ་པའོ། །། ༈ །སྤྱི་ཆིངས་ཀྱི་དགོངས་པ། ནམ་མཁའ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། དཔག་བསམ་ཤིང་གི་གྲིབ་མ་ལ། ཐོ་རེངས་ཁར་ཨཱརྻ་པ་ལོ་འདི་སྐད་ལོ། །བྱང་ཆུབ་སེམས་ནི་ནམ་མཁའ་ལྟ་བུའི་ཀློང་། །གསལ་ལ་རྟོག་པ་མེད་པའི་ཡེ་ཤེས་ནི། །ཕྱི་དང་ནང་མེད་ཀུན་ལ་གསལ་བར་སྣང་། །གདོད་ནས་དག་པ་ནམ་མཁའི་ཀློང་འདྲ་ལ། །རྣམ་རྟོག་བསྲེག་པའི་ཡེ་ཤེས་མངོན་ཤར་བས། །སྐྱེ་འགག་ཡངས་དོགས་རང་གཞན་མེད་པ་
༡༧༧ལས། །སངས་རྒྱས་དགོངས་པ་འདི་ལས་མེད་པར་མཇལ། །ཟ་ན་སེམས་ཉིད་བདུད་རྩི་ཟོ། །གོན་ན་ཆོས་ཉིད་གོས་སུ་གྱོན། །ཨེ་མ་ཀོ་ལ་ཤ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། ལུང་གི་རྒྱལ་པོ་ལས། སེམས་ཀྱི་དེ་ཉིད་མ་བཅོས་གསལ། །དེ་ལྟར་ཡིན་པས་རྟོག་པ་མེད། །ཅེས་སོ། །སྤྱི་ཆིངས་ཀྱི་དགོངས་བཅུད་དེ་བཅུ་གཅིག་པའོ། །། ༈ །བདེ་བ་འཕྲ་བཀོད་ཀྱི་དགོངས་པ། ཕྲིན་ལས་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། །རིན་པོ་ཆེའི་གླིང་དུ་ནམ་གུང་ཡོལ་ལ། ཨོ་རྒྱན་གྱི་མཁན་པོ་དྷ་ཧེ་འདི་སྐད་ལོ། སེམས་ཉིད་སངས་རྒྱས་ཡིན་པ་ལས། །སེམས་ལས་སངས་རྒྱས་ལོགས་ན་མེད། །སེམས་ཉིད་ཡེ་ནས་སངས་རྒྱས་ཡིན་པ་ལས། །རིག་པ་མི་འགྱུར་རང་བཞིན་མེད་པ་འདི། །ཟབ་མོའི་ཆོས་དབྱིངས་མི་བརྗོད་དོན་དམ་གསལ། །རྒྱ་མཚོ་ཆེ་ལྟར་ཆོས་དབྱིངས་ཟབ་མོ་ལས། །རིག་པ་མི་འགྱུར་བཟོད་ཅིང་ནང་ནས་འཆར། །ཀུན་གཞི་གཏན་ནས་སྦྱོང་བའི་ཉི་ཆེན་པོ། །མཁའ་ལྟར་དག་པས་ཡོངས་ལ་ཁྱབ་པར་བརྡལ། །ཆོས་ཉིད་ཡེ་ནས་དག་པས་སྤྱད་དུ་མེད་པ་ལ། །རིག་པ་མི་འགྱུར་གདོད་ནས་གསལ་བར་གནས། །ཕྱིན་ཅི་ལོག་གིས་མ་ལྷད་གསལ་བར་གནས། །བདེ་བར་གཤེགས་པའི་དགོངས་པ་འདི་ལས་མེད་པར་མཇལ། །ཆོས་ཉིད་ནམ་མཁའ་ཆེན་པོ་ལ། །རྣལ་འབྱོར་ཁྱུང་ཆེན་གཤོག་པ་ལྡིང་། །ཀརྨ་ཀོ་ལ་ཤ་ཨ་ཧོ༔་ཞེས་གསུངས་སོ། །གཞན་ཡང་། རྒྱ་ལུང་ཆེན་མོ་ལས། །སེམས་ཉིད་མི་རྟོག་ནམ་མཁའི་ངང་དུ་གནས། །དེས་ན་མ་བཅོས་རྟོག་མེད་ངང་དུ་ཤར། །ཞེས་སོ། །བདེ་བ་འཕྲ་བཀོད་ཀྱི་དགོངས་བཅུད་དེ་བཅུ་གཉིས་པའོ། །། ༈ །སྲོག་གི་འཁོར་ལོའི་དགོངས་པ། རང་བྱུང་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། ནྱ་གྲོ་དྷའི་དགོན་པར་སྲོད་ཡོལ་གྱི་
༡༧༨དུས་སུ། །དྲང་སྲོང་བྷི་ཤ་ཏི་འདི་སྐད་ལོ། །མ་བཅོས་མཁའ་ལ་གཟའ་སྐར་ཀུན་འཕྱོ་བས། །གདོད་ནས་དག་པས་མཁའ་ནི་མི་བརྗོད་དེ། །བྱང་ཆུབ་སེམས་ནི་མཁའ་ལྟར་དག་པ་ལ། །མཚན་མའི་རྣམ་རྟོག་སྒྱུ་མས་ཅི་བསླད་ཀྱང་། །རྣལ་འབྱོར་སེམས་ཉིད་དབྱིངས་ལ་འཕོ་འགྱུར་མེད། །ཆོས་དབྱིངས་བརྡལ་བ་ཆེན་པོའི་དཀྱིལ་འཁོར་ལ། །རིག་པ་ཡེ་ཤེས་དུ་མས་བཏབ་པ་ནི། །གསང་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་གསལ་བར་འཆར། །ཤར་བ་ཉིད་ཀྱི་དོན་དེ་མ་ཡེངས་ལས། །སངས་རྒྱས་རྣམས་ལ་དགོངས་པ་འདི་ལས་མེད། །རིག་པ་ཡེ་ཤེས་རླུང་ཆེན་གྱིས། །རྣམ་པར་རྟོག་པའི་མུན་པ་གཡེངས། །ཨ་ཀ་ཤ་ཀོ་ལ་མ་དྷ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། ཀློང་ཆེན་འཁྱིལ་བ་ཞེས་བྱ་བའི་མན་ངག་ལས། སེམས་ལ་ཐམས་ཅད་རྫོགས་པ་ལྷུན་གྲུབ་སྟེ། །བཅོས་སླད་དགག་སྒྲུབ་མེད་པ་ཡེ་ཤེས་གསལ། །ཅེས་སོ། །སྲོག་གི་འཁོར་ལོའི་དགོངས་བཅུད་དེ་བཅུ་གསུམ་པའོ། །། ༈ །རིན་ཆེན་ཀུན་འདུས་ཀྱི་དགོངས་པ། སྐྱེ་མེད་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། རྒྱས་པའི་དགོན་གནས་སུ་ནམ་སྨད་ལ་སྲས་མོ་གོ་མ་དེ་བཱི་འདི་སྐད་ལོ། །སངས་རྒྱས་ཀུན་གྱི་ཐུགས་ཉིད་ཡི་གེ་ཨ་ཡིན་ཏེ། །ཨ་ཡི་རང་བཞིན་ཡེ་ནས་དངོས་པོ་མེད། །བདེ་གཤེགས་ཀུན་གྱི་འབྱུང་གནས་ཨ་ཡིན་ཏེ། །བདེན་གཉིས་ལ་སོགས་སྣང་སྲིད་ཅིར་ཡང་སྣང་། །ཨ་ཡི་དངོས་གཞི་ཡེ་ནས་བྱང་ཆུབ་སེམས། །འཕོ་འགྱུར་འདུ་འབྲལ་བརྟགས་ནས་སངས་རྒྱས་ཏེ། །སྨྲ་བསམ་ཡུལ་ལས་འདས་པའི་བྱང་ཆུབ་དབྱིངས་ཉིད་ལ། །ཨ་ལྟར་སྐྱེ་བ་མེད་པའི་བྱང་ཆུབ་སེམས། །མ་བཅོས་མ་ཡེངས་མཉམ་པར་གཞག་པ་ལས། །བདག་གི་དགོངས་པ་འདི་ལས་མེད་པར་མཇལ། །ཆོས་ཉིད་དབྱེ་ཐང་ཆེན་པོ་ལ། །རིག་པའི་ཡེ་ཤེས་རྟ་ཕོ་རྒྱུག །ཀ་པ་ལ་ཀ་མ་པ་ཤ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། བདུད་རྩི་
༡༧༩འཁྱིལ་པ་ལས། །ཐམས་ཅད་ཉིད་དུ་མེད་པའི་བྱང་ཆུབ་སེམས། །ཡིན་པར་བདག་ཅག་རྣམས་ཀྱིས་རྟོགས། །ཞེས་སོ། །རིན་ཆེན་ཀུན་འདུས་ཀྱི་དགོངས་བཅུད་དེ་བཅུ་བཞི་པའོ། །། ༈ །སྐྱེ་མེད་རང་རིག་གི་དགོངས་པ། །ཆོས་སྐུ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། །ཙནྡན་སྡོང་པོའི་རྩར་གུང་ཐུན་ལ། །ཁ་ཆེའི་རྒྱལ་པོ་རབ་སྣང་འདི་སྐད་ལོ། །བྱང་ཆུབ་སེམས་ནི་ཡེ་ནས་ཆོས་ཀྱི་སྐུ། །མཁའ་དང་འདྲ་བར་ཀུན་ལ་ཁྱབ་པར་སྣང་། །སྣང་བ་ཉིད་ཀྱང་སྟོང་པ་སྟེ། །སྟོང་པའི་ཆོ་འཕྲུལ་ཅི་སྣང་བ། །བསམ་དུ་རྡུལ་ཙམ་མེད་པ་ལ། །བརྗོད་བྲལ་མེད་པར་མ་ཡེངས་ཞོག །ཆོས་ཉིད་སེམས་ཀྱི་བདུད་རྩི་ཡིས། །ཁམས་གསུམ་འགྲོ་བའི་ནད་པ་གསོས། །ཀ་ལ་པ་ཀོ་ཤ་ཨ་ཧོཿ ཞེས་གསུངས། གཞན་ཡང་། གླང་པོ་རབ་འབོགས་ལས། གཉིས་མེད་བདེ་ཆེན་ངང་གིས་གསལ། །དེ་བཞིན་མ་བཅོས་རྣལ་མར་ཞོག །ཅེས་སོ། །སྐྱེ་མེད་རང་རིག་གི་དགོངས་བཅུད་དེ་བཅོ་ལྔ་པའོ། ། ༈ བདེ་བ་རབ་འབྱམས་ཀྱི་དགོངས་པ། སྐུ་གསུམ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། གྲོང་ཁྱེར་དཔལ་ཡོན་དུ་སྲོད་ཀྱི་དུས་སུ་རྒྱལ་པོ་དེ་བ་རཱ་ཛ་འདི་སྐད་ལོ། སྣང་སྲིད་ཆོས་རྣམས་མངོན་དུ་ལྟོས་པ་རྣམས། །ཡང་དག་ཉིད་དུ་བཅོམ་པར་བྱ། །སྣང་སྲིད་ཐམས་ཅད་ཡེ་ནས་ཀུན་ཏུ་བཟང་པོའི་དང་། །འདི་ལས་གཞན་མེད་མ་ཡེངས་མཉམ་བཞག་པས། །བསྒོམ་པར་བྱ་བ་འདི་ལས་མེད་པར་རྟོགས། །ལུས་ནི་སྒྱུ་མའི་ཕུང་པོ་མིང་ཙམ་སྟེ། །སེམས་ནི་ཡེ་ཤེས་ཆེན་པོར་སོན། །ཙནྡྲ་དེ་བ་ཀོ་ས་ལ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། དྲི་མེད་བཤེས་གཉེན་གྱི་མན་ངག་ལས། དུག་ལྔ་ཡེ་ཤེས་ལྔ་རུ་གསལ། །དེ་ལྟར་རྟོགས་ན་རིག་མཆོག་ཡིན། །ཞེས་སོ། །བདེ་བ་རབ་འབྱམས་ཀྱི་དགོངས་བཅུད་དེ་བཅུ་དྲུག་པའོ། །
༡༨༠། ༈ །ནམ་མཁའ་རྒྱལ་པོའི་དགོངས་པ། གྲོལ་བ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། མེ་ཏོག་གི་ཚལ་དུ་ནམ་གྱི་ཆ་སྟོད་ལ། དྷརྨ་རཱ་ཛ་འདི་སྐད་དོ། །སེམས་ཀྱི་རང་བཞིན་ཡེ་ནས་མཛོད་ཆེན་ལས། །གང་ལྟར་རྟོགས་ཤིང་ཇི་ལྟར་སྣང་བ་ཀུན། །འབྱུང་འཇུག་འབྱུང་བ་མེད་ན་ཀུན་ཏུ་བཟང་། །གཉིས་སུ་མེད་པའི་རིག་པ་དེ། །ཀུན་ཏུ་བཟང་པོའི་སེམས་ལ་གནས། །རིག་པའི་དབྱིངས་ལས་མ་བསླད་པས། །སྒོམ་ཞིང་གནས་པ་འདི་ལ་མེད། །ཀྱེ་མ་མ་རྟོགས་སྙིང་རེ་རྗེ། །བདག་ཉིད་རང་སར་གྲོལ་བ་ཨ་ལ་ལ། །དྷོ་ཧ་ཀོ་པ་ཤ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། འཕགས་པ་དྲང་སྲོང་མིག་གི་མན་ངག་ལས། མ་ཡེངས་རྩེ་གཅིག་བསམ་དང་བྲལ། །དེས་ན་ཡེ་ཤེས་ཟང་ཐལ་ཡིན། །ཅེས་སོ། །ནམ་མཁའ་རྒྱལ་པོའི་དགོངས་བཅུད་དེ་བཅུ་བདུན་པའོ། །། ༈ །རྗེ་བཙུན་དམ་པའི་དགོངས་པ། ཤེས་རབ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། དབེན་པ་ཆེན་པོའི་གནས་སུ་བུདྡྷ་གུཔྟ་འདི་སྐད་ལོ། །ཆོས་རྣམས་ཐམས་ཅད་མ་བཅོས་ཀུན་ཏུ་བཟང་པོ་ལ། །གསལ་ལ་འགྱུར་པ་མེད་པ་བྱང་སེམས་ངང་དེ་ལ། །མཚན་མའི་རྣམ་པར་རྟོག་པས་མ་བསླད་དེ། །རྣམ་རྟོག་རང་བྱུང་རང་ཞིར་གནས་པ་ལས། །སྒོམ་པ་འདི་ལས་གུད་ན་མེད་པར་འཁུམས། །སེམས་ཀྱི་མེ་ལོང་མ་བལྟས་ན། །རྣམ་རྟོག་དྲི་མས་གོས་ཏ་རེ། །པྲཛྙཱ་ཀོ་ས་ལ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །གཞན་ཡང་། །སྒོམ་ལུང་ཆེན་མོ་ལས། བཅོས་བསླད་མེད་པར་ཀ་ནས་དག་པ་ལ། །གཉིས་སུ་འཛིན་པ་ཡང་ཡེ་ངང་གིས་མེད། །ཅེས་སོ། །རྗེ་བཙན་དམ་པའི་དགོངས་བཅུད་དེ་བཅོ་བརྒྱད་པའོ། །། ༈ །སེམས་སྡེ་སྤྱིའི་དགོངས་པ། རིན་པོ་ཆེ་ཀློང་རྡོལ་གླུའི་བསམ་གཏན། ཤྲཱི་སིདྷས་བསམ་གཏན་གྱི་ཁང་པར་ཚོགས་ཀྱི་མཆོད་པའི་དུས་སུ་བླངས་པ་འདི་སྐད་ལོ།
༡༨༡ཇི་ལྟར་བསམ་ཀྱང་སྒོམ་དུ་མེད། །ཇི་ལྟར་སྨྲས་ཀྱང་དཔྱད་དུ་མེད། །ཡེ་ཤེས་ཆེན་པོའི་དབྱིངས་ཉིད་ལས། །བཙལ་བ་མེད་པར་འོད་ཤར་ཏེ། །བསམ་དུ་མེད་པའི་ཆོས་ཉིད་ལ། །ཡེངས་པ་མེད་པར་མཉམ་པར་གཞག །སྒོམ་པ་འདི་ལས་མེད་པར་འཁུམས། །རིག་པ་ཡེ་ཤེས་མདུང་རྩེ་ཡིས། །རྣམ་རྟོག་དགྲ་བོ་རྡུལ་དུ་རློག །རཏྣ་ཀོ་ས་ལ་ཨ་ཧོཿ ཞེས་གསུངས་སོ། །སེམས་སྡེ་སྤྱིའི་དགོངས་བཅུད་དེ་བཅུ་དགུ་པའོ། །། ༈ །རྫོགས་པ་ཆེན་པོ་སེམས་སྡེ་བཅོ་བརྒྱད་ཀྱི་དགོངས་བཅུད་རིག་འཛིན་གྲུབ་ཐོབ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུར་བཞེངས་པ་རྫོགས་སོ། ཤུ་བྷཾ།
